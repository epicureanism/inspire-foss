# basically: capture a getcapabilities document once and put it in static file
# (e.g. /content/wfscaps.xml). Put deegree WFS behind Apache and intercept
# getcapabilities using the mod_rewrite config like below
# see also http://corz.org/serv/tricks/htaccess2.php
Options +FollowSymlinks
RewriteEngine On
RewriteCond %{QUERY_STRING} request=GetCapabilities
RewriteRule ^/deegree3/services(.*) /content/wfscaps.xml
